## Persp
x <- seq(-10, 10, length= 30)
y <- x
f <- function(x,y) { r <- sqrt(x^2+y^2); 10 * sin(r)/r }
z <- outer(x, y, f)
z[is.na(z)] <- 1
# 0.5 to include z axis label
par(mar=c(0, 0.5, 0, 0), lwd=0.5)
persp(x, y, z, theta = 30, phi = 30, 
      expand = 0.5)
par(mar=c(5.1, 4.1, 4.1, 2.1), lwd=1)

Prepare sorted values for x (life_exp) and y (gdp)

life_exp_sorted <- sort(unique(HPI$`Life Expectancy (years)`))
gdp_sorted <- sort(unique(HPI$`GDP per capita ($)`))

#### Create a grid of x (life_exp) and y (gdp)
life_exp_grid <- seq(min(life_exp_sorted), max(life_exp_sorted), length.out = 50)
gdp_grid <- seq(min(gdp_sorted), max(gdp_sorted), length.out = 50)

### Interpolate z (hpi) values into a matrix form that corresponds to the grid. We will use a simple interpolation method here for the sake of the 3D plot
hpi_matrix <- outer(life_exp_grid, gdp_grid, FUN = function(x, y) {
  #### Use a simple linear model or any function relating life expectancy and GDP to HPI
  #### For simplicity, let's assume HPI increases with both life expectancy and GDP
  return((x / max(life_exp_grid)) * (y / max(gdp_grid)) * max(HPI$HPI, na.rm = TRUE))
})

## Create the 3D perspective plot
persp(life_exp_grid, gdp_grid, hpi_matrix, 
      theta = 30, phi = 20, 
      col = "lightblue", 
      xlab = "Life Expectancy (years)", 
      ylab = "GDP per Capita ($)", 
      zlab = "HPI",
      main = "3D Perspective of Life Expectancy, GDP, and HPI")
